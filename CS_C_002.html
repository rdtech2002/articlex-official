16. A program in C to display the sum of the series [ 1+x+x2/2! +x3/3! +....] upto nth term, where value of n and x should be provided by the user.

#include
#include
#include
int factorial(float n)
{
float m = 1;
if (n == 0)
return 1;
while (n > 0)
{
m = m * n;
n--;
}
return m;
}
int main()
{
float n, x;
printf("Enter number of terms:\n");
scanf("%f", &n);
printf("\nThe sample series is:\n");
printf("1 + x + (x^2)/2! + (x^3)/3! + ...");
printf("\nEnter value of 'x' in the above series:\n");
scanf("%f", &x);
printf("\nThe required series is:\n");
float fac, power, num, sum;
for (float i = 0; i < n; i++)
{
fac = factorial(i);
power = pow(x, i);
num = power / fac;
if (i != n)
printf("%.0f/%.0f, ", power, fac);
else
printf("%.0f/%.0f. ", power, fac);
sum = sum + num;
}
printf("\nThe sum of this series is: %.3f", sum);
return 0;
}

INPUT

Enter number of terms:
4

The sample series is:
1 + x + (x^2)/2! + (x^3)/3! + ...
Enter value of 'x' in the above series:
2

The required series is:
1/1, 2/1, 4/2, 8/6,
The sum of this series is: 6.333 

OUTPUT
